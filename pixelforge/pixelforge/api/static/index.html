CLI quickstart

# 1) Analyze a project
python -m pixelforge.cli analyze --image ./input.jpg

# 2) Segment with SAM (auto mode)
python -m pixelforge.cli segment auto --image ./input.jpg -c sam_vit_b.pth

# 3) Run the Web UI
uvicorn pixelforge.api.main:app --port 8700 --reload
